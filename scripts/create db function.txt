def create_database():
    conn = sqlite3.connect('f1_racing.db')
    cursor = conn.cursor()
    
    # Create tables
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS drivers (
            DriverID INTEGER PRIMARY KEY,
            Name TEXT,
            Team TEXT,
            Nationality TEXT,
            Age INTEGER
        )
    ''')
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS historical_wins (
            DriverID INTEGER,
            Year INTEGER,
            Wins INTEGER,
            FOREIGN KEY (DriverID) REFERENCES drivers(DriverID)
        )
    ''')
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS tracks (
            TrackID INTEGER PRIMARY KEY,
            TrackName TEXT,
            Location TEXT,
            Length_km REAL,
